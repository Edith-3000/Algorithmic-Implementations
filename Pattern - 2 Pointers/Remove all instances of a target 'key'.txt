long long removeKeyInPlace(vector<long long> &v, long long key)
{
	//i stores the index(of last number(which is not
	//the key)seen so far
	long long i=0;
	//to iterate over v[]
	long long j=0;

	for(; j<v.size(); j++)
	{
	   if(v[i]!=key)
	   {
	      v[i]=v[j];
	      i++;
	   }
	}

	return i;
}
